async function login() {

  const email = document.getElementById('auth-email').value.trim();

// HELPERS

function timeAgo(ts) {

  if (!ts) return '—';

  const diff = (Date.now() - new Date(ts).getTime()) / 1000;

  if (diff < 60) return 'hace ' + Math.round(diff) + 's';

  if (diff < 3600) return 'hace ' + Math.round(diff/60) + 'min';

  if (diff < 86400) return 'hace ' + Math.round(diff/3600) + 'h';

  return 'hace ' + Math.round(diff/86400) + 'd';

}


function statClass(val, warn=70, danger=85) {

  if (!val && val !== 0) return '';

  if (val >= danger) return 'danger';

  if (val >= warn) return 'warn';

  return '';

}


function statStr(val) {

  if (val === null || val === undefined) return '—';

  return Math.round(val) + '%';

}


function isOnline(ts) {

  if (!ts) return false;

  return (Date.now() - new Date(ts).getTime()) < 10 * 60 * 1000;

}


function serverRow(s, extended=false) {

  const online = isOnline(s.ultimo_heartbeat);

  const estado = online ? 'online' : 'offline';

  const estadoLabel = online ? 'Online' : 'Offline';

  const cpu = s.cpu_last; const ram = s.ram_last; const disco = s.disco_last;


  let row = `<td><div class="server-name">${s.nombre}</div><div class="server-id">${s.id}</div></td>`;

  row += `<td style="font-size:13px;color:var(--text-2)">${s.instalador_nombre || '—'}</td>`;

  row += `<td><span class="badge ${estado}"><span class="badge-dot"></span>${estadoLabel}</span></td>`;

  row += `<td><span class="stat ${statClass(cpu)}">${statStr(cpu)}</span></td>`;

  row += `<td><span class="stat ${statClass(ram)}">${statStr(ram)}</span></td>`;

  row += `<td><span class="stat ${statClass(disco)}">${statStr(disco)}</span></td>`;

  if (extended) {

    row += `<td style="font-size:11px;font-family:'DM Mono',monospace;color:var(--text-3)">${s.ha_version || '—'}</td>`;

    row += `<td style="font-size:12px;color:var(--text-3)">${timeAgo(s.ultimo_heartbeat)}</td>`;

  }

  row += `<td><button class="action-btn" onclick="connectServer('${s.id}','${s.nombre}')">Conectar</button></td>`;

  return '<tr>' + row + '</tr>';

}


async function getServersWithStats() {

  let query = sb.from('servidores').select('*').eq('activo', true);

  if (currentProfile.rol === 'instalador') query = query.eq('instalador_id', currentProfile.id);

  const { data: servers } = await query.order('creado_en', { ascending: false });

  if (!servers) return [];


  const { data: installers } = await sb.from('usuarios').select('id,nombre').in('rol', ['instalador','admin','superadmin']);

  const instMap = {};

  if (installers) installers.forEach(i => instMap[i.id] = i.nombre);


  const { data: hbs } = await sb.from('heartbeats').select('servidor_id,cpu,ram,disco,recibido_en').order('recibido_en', { ascending: false });

  const hbMap = {};

  if (hbs) hbs.forEach(h => { if (!hbMap[h.servidor_id]) hbMap[h.servidor_id] = h; });


  return servers.map(s => ({

    ...s,

    instalador_nombre: instMap[s.instalador_id] || '—',

    cpu_last: hbMap[s.id]?.cpu ?? null,

    ram_last: hbMap[s.id]?.ram ?? null,

    disco_last: hbMap[s.id]?.disco ?? null,

  }));

}


// DASHBOARD

async function loadDashboard() {

  const servers = await getServersWithStats();

  const online = servers.filter(s => isOnline(s.ultimo_heartbeat)).length;

  const offline = servers.length - online;


  document.getElementById('m-total').textContent = servers.length;

  document.getElementById('m-online').textContent = online;

  document.getElementById('m-offline').textContent = offline;


  if (currentProfile.rol === 'superadmin') {

    const { count } = await sb.from('usuarios').select('*', {count:'exact',head:true}).eq('rol','instalador').eq('activo',true);

    document.getElementById('m-installers').textContent = count || 0;

  }


  const tbody = document.getElementById('dashboard-servers-body');

  if (!servers.length) { tbody.innerHTML = '<tr><td colspan="7" class="empty">No hay servidores registrados aún</td></tr>'; return; }

  tbody.innerHTML = servers.slice(0,6).map(s => serverRow(s)).join('');

}


// SERVERS

async function loadServers() {

  const tbody = document.getElementById('servers-body');

  tbody.innerHTML = '<tr><td colspan="9" class="loading">Cargando...</td></tr>';

  const servers = await getServersWithStats();

  tbody.innerHTML = servers.map(s => serverRow(s, true)).join('');

  await sb.from('usuarios').update({ activo: !activo }).eq('id', id);

}


// ALERTS

async function loadAlertBadge() {

  const { data: hbs } = await sb.from('heartbeats').select('servidor_id,cpu,ram,disco,recibido_en').order('recibido_en', { ascending: false });

  if (!hbs) return;

  const hbMap = {};

  hbs.forEach(h => { if (!hbMap[h.servidor_id]) hbMap[h.servidor_id] = h; });

  let count = 0;

  Object.values(hbMap).forEach(h => {

    if ((h.cpu||0) > 80 || (h.ram||0) > 80 || (h.disco||0) > 80) count++;

  });

  const badge = document.getElementById('alert-badge');

  if (count > 0) { badge.style.display = 'inline'; badge.textContent = count; }

}


async function loadAlerts() {

  const tbody = document.getElementById('alerts-body');

  const servers = await getServersWithStats();

  const alerts = [];


  servers.forEach(s => {

    if (!isOnline(s.ultimo_heartbeat)) {

      alerts.push({ servidor: s.nombre, tipo: 'offline', detalle: 'Sin conexión', cuando: s.ultimo_heartbeat });

    } else {

      if ((s.cpu_last||0) > 80) alerts.push({ servidor: s.nombre, tipo: 'warn', detalle: `CPU al ${Math.round(s.cpu_last)}%`, cuando: s.ultimo_heartbeat });

      if ((s.ram_last||0) > 80) alerts.push({ servidor: s.nombre, tipo: 'warn', detalle: `RAM al ${Math.round(s.ram_last)}%`, cuando: s.ultimo_heartbeat });

      if ((s.disco_last||0) > 80) alerts.push({ servidor: s.nombre, tipo: 'warn', detalle: `Disco al ${Math.round(s.disco_last)}%`, cuando: s.ultimo_heartbeat });

    }

  });


  if (!alerts.length) { tbody.innerHTML = '<tr><td colspan="4" class="empty">✓ Sin alertas activas</td></tr>'; return; }

  tbody.innerHTML = alerts.map(a => `

    <tr>

      <td style="font-weight:500">${a.servidor}</td>

      <td><span class="badge ${a.tipo === 'offline' ? 'offline' : 'warn'}">${a.tipo === 'offline' ? 'Offline' : 'Alerta'}</span></td>

      <td style="color:var(--text-2)">${a.detalle}</td>

      <td style="font-size:12px;color:var(--text-3)">${timeAgo(a.cuando)}</td>

    </tr>

  `).join('');

}

async function openInviteModal() {

  const token = crypto.randomUUID().replace(/-/g,'').slice(0,16);

  inviteToken = token;

  const expira = new Date(Date.now() + 5 * 60 * 1000);


  await sb.from('invitaciones').insert({

    token,

    creado_por: currentProfile.id,

    expira_en: expira.toISOString()

  });


  const link = `${location.origin}/register?invite=${token}`;

  document.getElementById('invite-link-text').textContent = link;

  document.getElementById('invite-modal').classList.add('open');


  let secs = 300;

  clearInterval(inviteCountdownInterval);

  updateCountdown(secs);

  inviteCountdownInterval = setInterval(() => {

    secs--;

    if (secs <= 0) { clearInterval(inviteCountdownInterval); closeModal(); showToast('Enlace expirado'); return; }

    updateCountdown(secs);

  }, 1000);

}


function updateCountdown(secs) {

  const m = Math.floor(secs/60).toString().padStart(2,'0');

  const s = (secs%60).toString().padStart(2,'0');

  document.getElementById('invite-countdown').textContent = `Expira en ${m}:${s}`;

}


function closeModal() {

  document.getElementById('invite-modal').classList.remove('open');

  clearInterval(inviteCountdownInterval);

}


function copyInviteLink() {

  const text = document.getElementById('invite-link-text').textContent;

  navigator.clipboard.writeText(text).then(() => showToast('Enlace copiado'));

}
